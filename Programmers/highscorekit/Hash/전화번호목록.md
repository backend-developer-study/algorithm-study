import java.util.*;

class Solution {
    public boolean solution(String[] phone_book) {
        Arrays.sort(phone_book);
        Set<String> dict = new HashSet<>();

        boolean answer = true;

        for (String phoneNum:phone_book) {
            if (!answer)break;
            for (String preNumber:dict) {
                if(phoneNum.startsWith(preNumber)) {
                    answer = false;
                    break;
                }
            }
            dict.add(phoneNum);
        }
        return answer;
    }
}
